	if(score>=A)
	  printf("Student has an A grade\n");
	else if(score>=B)
	  printf("Student has an B grade\n");
	else if(score>=C)
	  printf("Student has an C grade\n");
	else if(score>=D)
	  printf("Student has an D grade\n");
	else if(score<D)
	  printf("Student has failed the course\n");

	return 0;
